		int lower_one_mask(int n) {
		 int a = -1; // Initialize with all bits set to 1		 
		 // Shift happens in two steps to handle the case n = word size.
		 a = a << n-1;
		 a = a << 1;		 
		 return a ^ -1; // Complement by xoring with -1
		}
		\end{verbatim}
